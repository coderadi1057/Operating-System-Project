#include<stdio.h>
#include<unistd.h>
int main()
{
	int n,i,c,t,flag=0;
	printf("\n Enter number of process");
	scanf("%d",&n);
	int bt[n],at[n],tq=3,p[n],tat=0,wt=0,rt[n],r=n;
	for(i=0;i<n;i++){
		p[i]=i;
		printf("\nEnter Arrival Time of p[%d] process",i+1);
		scanf("%d",&at[i]);
		printf("Enter Burst Time of p[%d] process",i+1);
		scanf("%d",&bt[i]);
		rt[i]=bt[i];
	}
	printf("\nProcess\t|Turnaround Time|Waiting Time\n");
	for(t=0,c=0;r!=0;)
	{
		if(rt[c]<=tq&&rt[c]>0)
		{
			t=t+rt[c];
			rt[c]=0;
			flag=1;
		}
		else if(rt[c]>0)
		{
		rt[c]-=tq;
		t=t+tq;
		}
		if(rt[c]==0&&flag==1)
		{
		r--;
		printf("P[%d]\t|\t%d\t|\t%d\n",c+1,t-at[c],t-at[c]-bt[c]);
		wt+=t-at[c]-bt[c];
		tat+=t-at[c];
		flag=0;
		}
		if(c==n-1)
		c=0;
		else if(at[c+1]<=t)
		c++;
		else
		c=0;
	}
printf("\nAverage Waiting Time= %f\n",wt*1.0/n);
printf("Avg Turnaround Time = %f",tat*1.0/n);
}
